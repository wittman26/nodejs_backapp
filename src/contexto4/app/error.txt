SELECT
    TS.TRADE_SIGNATURE_ID,
    TS.EXPEDIENT_ID,
    SG.SIGNER_ID,
    SG.DOCUMENT_TYPE,
    SG.DOCUMENT_NUMBER,
    SG.NAME,
    SG.IS_CLIENT,
    SG.INTERVENTION_TYPE,
    COALESCE(DD.GN_ID, '-') AS GN_ID,
    DD.TYPE_DOC AS DOCUMENTAL_TYPE_DOC,
    DD.INDICATOR_PRECONTRACTUAL,
    DS.SIGNED_DOC,
    DS.SIGN_DATE,
	ds.*
FROM ACELER_FX.FX_TRADE_SIGNATURE TS
         LEFT JOIN ACELER_FX.FX_TRADE_SIGNER SG
                   ON TS.TRADE_SIGNATURE_ID = SG.TRADE_SIGNATURE_ID
         LEFT JOIN ACELER_DFD.DFD_EXPEDIENT DE
                   ON TS.EXPEDIENT_ID = DE.ID_EXP_SEC
         LEFT JOIN ACELER_DFD.DFD_EXPEDIENT_DOCUMENT DED
              ON DED.EXPEDIENT_ID = DE.ID
         LEFT JOIN ACELER_DFD.DFD_DOCUMENT DD
              ON DED.DOCUMENT_ID = DD.ID
         LEFT JOIN (
    SELECT *
    FROM (
        SELECT DS.*,
               ROW_NUMBER() OVER (PARTITION BY DS.DOCUMENT_ID ORDER BY DS.SIGN_DATE DESC) AS RN
        FROM ACELER_DFD.DFD_SIGNER DS
    )
    WHERE RN = 1
) DS ON DD.ID = DS.DOCUMENT_ID;
